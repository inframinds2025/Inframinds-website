        // Simple mobile menu toggle
        document.querySelector('.mobile-menu').addEventListener('click', function() {

            navLinks.style.display = navLinks.style.display === 'flex' ? 'none' : 'flex';
        });

        // Smooth scrolling for anchor links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                
                const targetId = this.getAttribute('href');
                if(targetId === '#') return;
                
                const targetElement = document.querySelector(targetId);
                if(targetElement) {
                    window.scrollTo({
                        top: targetElement.offsetTop - 80,
                        behavior: 'smooth'
                    });
                    
                    // Close mobile menu if open
                    if(window.innerWidth <= 768) {
                        document.querySelector('.nav-links').style.display = 'none';
                    }
                }
            });
        });
            // Consultation Form Functionality
        class ConsultationForm {
            constructor() {
                this.currentStep = 1;
                this.totalSteps = 4;
                this.formData = {};
                this.init();
            }

            init() {
                this.bindEvents();
                this.showStep(1);
            }

            bindEvents() {
                // Next button clicks
                document.querySelectorAll('.btn-next').forEach(btn => {
                    btn.addEventListener('click', () => this.nextStep());
                });

                // Previous button clicks
                document.querySelectorAll('.btn-prev').forEach(btn => {
                    btn.addEventListener('click', () => this.prevStep());
                });

                // Form submission
                const form = document.getElementById('consultationForm');
                if (form) {
                    form.addEventListener('submit', (e) => {
                        e.preventDefault();
                        this.submitForm();
                    });
                }

                // New request button
                const newRequestBtn = document.getElementById('newRequest');
                if (newRequestBtn) {
                    newRequestBtn.addEventListener('click', () => {
                        this.resetForm();
                    });
                }

                // Real-time validation
                this.addRealTimeValidation();
            }

            showStep(step) {
                // Hide all steps
                document.querySelectorAll('.form-step').forEach(stepEl => {
                    stepEl.classList.remove('active');
                });

                // Show current step
                const currentStepEl = document.querySelector(`.form-step[data-step="${step}"]`);
                if (currentStepEl) {
                    currentStepEl.classList.add('active');
                }

                // Update progress indicator
                document.querySelectorAll('.step').forEach(stepEl => {
                    stepEl.classList.remove('active');
                });
                const progressStep = document.querySelector(`.step[data-step="${step}"]`);
                if (progressStep) {
                    progressStep.classList.add('active');
                }

                this.currentStep = step;

                // Update review on step 4
                if (step === 4) {
                    this.updateReview();
                }
            }

            nextStep() {
                if (this.validateStep(this.currentStep)) {
                    this.saveStepData(this.currentStep);
                    if (this.currentStep < this.totalSteps) {
                        this.showStep(this.currentStep + 1);
                    }
                }
            }

            prevStep() {
                if (this.currentStep > 1) {
                    this.showStep(this.currentStep - 1);
                }
            }

            validateStep(step) {
                let isValid = true;
                const stepEl = document.querySelector(`.form-step[data-step="${step}"]`);
                if (!stepEl) return false;

                // Clear previous errors
                stepEl.querySelectorAll('.error').forEach(el => {
                    el.classList.remove('error');
                });

                // Validate required fields
                const requiredFields = stepEl.querySelectorAll('[required]');
                requiredFields.forEach(field => {
                    if (!field.value.trim()) {
                        this.showError(field, 'This field is required');
                        isValid = false;
                    } else if (field.type === 'email' && !this.isValidEmail(field.value)) {
                        this.showError(field, 'Please enter a valid email address');
                        isValid = false;
                    } else if (field.type === 'tel' && !this.isValidPhone(field.value)) {
                        this.showError(field, 'Please enter a valid phone number');
                        isValid = false;
                    }
                });

                // Validate checkboxes (challenges)
                if (step === 2) {
                    const challenges = stepEl.querySelectorAll('input[name="challenges"]:checked');
                    if (challenges.length === 0) {
                        const checkboxGroup = stepEl.querySelector('.checkbox-group');
                        if (checkboxGroup) {
                            this.showError(checkboxGroup, 'Please select at least one challenge');
                        }
                        isValid = false;
                    }
                }

                return isValid;
            }

            showError(field, message) {
                const formGroup = field.closest('.form-group') || field.parentElement;
                formGroup.classList.add('error');
                const errorEl = formGroup.querySelector('.error-message');
                if (errorEl) {
                    errorEl.textContent = message;
                }
            }

            isValidEmail(email) {
                const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
                return emailRegex.test(email);
            }

            isValidPhone(phone) {
                const phoneRegex = /^[\+]?[1-9][\d]{0,15}$/;
                return phoneRegex.test(phone.replace(/[\s\-\(\)]/g, ''));
            }

            saveStepData(step) {
                const stepEl = document.querySelector(`.form-step[data-step="${step}"]`);
                if (!stepEl) return;

                const formData = new FormData(stepEl);
                
                for (let [key, value] of formData.entries()) {
                    if (key === 'challenges') {
                        if (!this.formData.challenges) this.formData.challenges = [];
                        this.formData.challenges.push(value);
                    } else {
                        this.formData[key] = value;
                    }
                }
            }

            updateReview() {
                const reviewContent = document.getElementById('reviewContent');
                if (!reviewContent) return;

                const fields = {
                    'Personal Information': [
                        { label: 'Name', value: `${this.formData.firstName} ${this.formData.lastName}` },
                        { label: 'Email', value: this.formData.email },
                        { label: 'Phone', value: this.formData.phone },
                        { label: 'Company', value: this.formData.company || 'Not provided' },
                        { label: 'Job Title', value: this.formData.jobTitle || 'Not provided' }
                    ],
                    'Project Details': [
                        { label: 'Primary Interest', value: this.getDisplayValue('primaryInterest', this.formData.primaryInterest) },
                        { label: 'Project Type', value: this.getDisplayValue('projectType', this.formData.projectType) },
                        { label: 'Current Infrastructure', value: this.formData.currentInfrastructure || 'Not provided' },
                        { label: 'Challenges', value: (this.formData.challenges || []).map(c => this.getDisplayValue('challenges', c)).join(', ') }
                    ],
                    'Logistics': [
                        { label: 'Timeline', value: this.getDisplayValue('timeline', this.formData.timeline) },
                        { label: 'Budget', value: this.getDisplayValue('budget', this.formData.budget) },
                        { label: 'Referral Source', value: this.getDisplayValue('referralSource', this.formData.referralSource) || 'Not provided' },
                        { label: 'Contact Method', value: this.getDisplayValue('contactMethod', this.formData.contactMethod) }
                    ]
                };

                let html = '';
                for (const [section, items] of Object.entries(fields)) {
                    html += `<h4>${section}</h4>`;
                    items.forEach(item => {
                        html += `
                            <div class="review-item">
                                <span class="review-label">${item.label}:</span>
                                <span class="review-value">${item.value}</span>
                            </div>
                        `;
                    });
                }

                reviewContent.innerHTML = html;
            }

            getDisplayValue(fieldName, value) {
                const valueMap = {
                    primaryInterest: {
                        'aws-setup': 'AWS Infrastructure Setup',
                        'devops': 'DevOps Implementation',
                        'cloud-migration': 'Cloud Migration',
                        'security': 'Security & Compliance',
                        'cost-optimization': 'Cost Optimization',
                        'other': 'Other'
                    },
                    projectType: {
                        'new-project': 'New Project/Startup',
                        'existing-improvement': 'Existing System Improvement',
                        'emergency-support': 'Emergency Technical Support',
                        'consulting': 'Consulting & Strategy'
                    },
                    challenges: {
                        'high-costs': 'High Cloud Costs',
                        'security': 'Security Concerns',
                        'performance': 'Performance Issues',
                        'scaling': 'Scaling Problems',
                        'expertise': 'Lack of In-house Expertise'
                    },
                    timeline: {
                        'immediate': 'Immediate (within 2 weeks)',
                        'short-term': 'Short-term (1-3 months)',
                        'planning': 'Planning phase (3-6 months)',
                        'future': 'Future consideration'
                    },
                    budget: {
                        'under-5k': 'Under $5,000',
                        '5k-15k': '$5,000 - $15,000',
                        '15k-30k': '$15,000 - $30,000',
                        '30k-50k': '$30,000 - $50,000',
                        '50k-plus': '$50,000+',
                        'discuss': 'To be discussed'
                    },
                    referralSource: {
                        'google': 'Google Search',
                        'linkedin': 'LinkedIn',
                        'referral': 'Referral',
                        'previous-client': 'Previous Client',
                        'event': 'Industry Event',
                        'other': 'Other'
                    },
                    contactMethod: {
                        'email': 'Email',
                        'phone': 'Phone Call',
                        'video': 'Video Call'
                    }
                };

                return valueMap[fieldName]?.[value] || value || 'Not provided';
            }

            addRealTimeValidation() {
                // Add input event listeners for real-time validation
                document.querySelectorAll('input, select, textarea').forEach(field => {
                    field.addEventListener('blur', () => {
                        if (field.hasAttribute('required')) {
                            this.validateField(field);
                        }
                    });

                    field.addEventListener('input', () => {
                        // Clear error when user starts typing
                        const formGroup = field.closest('.form-group');
                        if (formGroup && formGroup.classList.contains('error')) {
                            formGroup.classList.remove('error');
                        }
                    });
                });
            }

            validateField(field) {
                const formGroup = field.closest('.form-group');
                if (formGroup) {
                    formGroup.classList.remove('error');
                }

                if (field.hasAttribute('required') && !field.value.trim()) {
                    this.showError(field, 'This field is required');
                    return false;
                }

                if (field.type === 'email' && field.value && !this.isValidEmail(field.value)) {
                    this.showError(field, 'Please enter a valid email address');
                    return false;
                }

                if (field.type === 'tel' && field.value && !this.isValidPhone(field.value)) {
                    this.showError(field, 'Please enter a valid phone number');
                    return false;
                }

                return true;
            }

            async submitForm() {
                if (this.validateStep(4)) {
                    this.saveStepData(4);
                    
                    // Show loading state
                    const submitBtn = document.querySelector('.btn-submit');
                    if (submitBtn) {
                        const originalText = submitBtn.textContent;
                        submitBtn.textContent = 'Submitting...';
                        submitBtn.disabled = true;

                        try {
                            // Simulate API call
                            await new Promise(resolve => setTimeout(resolve, 1500));
                            
                            // Show success message
                            const form = document.getElementById('consultationForm');
                            const successMessage = document.getElementById('successMessage');
                            if (form && successMessage) {
                                form.style.display = 'none';
                                successMessage.style.display = 'block';
                            }
                            
                            // Here you would typically send the data to your backend
                            console.log('Form submitted:', this.formData);
                            
                        } catch (error) {
                            alert('There was an error submitting your request. Please try again.');
                            console.error('Submission error:', error);
                        } finally {
                            submitBtn.textContent = originalText;
                            submitBtn.disabled = false;
                        }
                    }
                }
            }

            resetForm() {
                const form = document.getElementById('consultationForm');
                const successMessage = document.getElementById('successMessage');
                if (form && successMessage) {
                    form.reset();
                    form.style.display = 'block';
                    successMessage.style.display = 'none';
                    this.formData = {};
                    this.showStep(1);
                }
            }
        }

        // Initialize the form when DOM is loaded
        document.addEventListener('DOMContentLoaded', () => {
            new ConsultationForm();
        });
    </script>
